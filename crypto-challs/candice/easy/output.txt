def egcd(a, b):
    if a == 0:
        return (b, 0, 1)
    else:
        g, y, x = egcd(b % a, a)
        return (g, x - (b // a) * y, y)

def sqrt(c):
    g ,yp, yq = egcd(p, q)
    mp = pow(c, (p+1)//4, p)
    mq = pow(c, (q+1)//4, q)
    r = (yp * p * mq + yq * q * mp) % n
    mr = n - r
    s = (yp * p * mq - yq * q * mp) % n
    ms = n - s
    print(f"Roots found: {r}, {mr}, {s}, {ms}")
    return [r, mr, s, ms]
